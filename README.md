        case 1:
            bilangan2 = Double.parseDouble(String.valueOf(txthasil.getText()));
            jumlah = bilangan1+bilangan2;
            bilangan = Double.toString(jumlah);            
            break;
        case 2:
            bilangan2 =Double.parseDouble(String.valueOf(txthasil.getText()));
            jumlah = bilangan1 - bilangan2;
            bilangan = Double.toString(jumlah);            
            break;
        case 3:
            bilangan2 = Double.parseDouble(String.valueOf(txthasil.getText()));
            jumlah = bilangan1 * bilangan2;
            bilangan = Double.toString(jumlah);            
            break;
        case 4:
            bilangan2 = Double.parseDouble(String.valueOf(txthasil.getText()));
            jumlah = bilangan1 / bilangan2;
            bilangan = Double.toString(jumlah);            
            break;            
        } 
        txthasil.setText(bilangan);
    }
